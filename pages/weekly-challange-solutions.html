c++ #include <bits/stdc++.h> using namespace std; unordered_map<int, int> dsu; unordered_map<int, int> dsu_sz; int mv = 0; bool hasp(int x){     return dsu.count(x); } int findp(int x){     if(dsu.count(x) == 0){         dsu[x] = x;         dsu_sz[x] = 1;         return x;     }     if(dsu[x] == x) return x;     return dsu[x] = findp(dsu[x]); } void merge(int x, int y){     x = findp(x);     y = findp(y);     if(x != y){         dsu_sz[x] += dsu_sz[y];         mv = max(mv, dsu_sz[x]);         dsu[y] = x;     } } int solve(vector<int>& input){     mv = 0;     for(int i = 0; i < input.size(); i++){         int x = input[i];         findp(x);         if(hasp(x-1)){             merge(x-1, x);         }         if(hasp(x+1)){             merge(x+1, x);         }     }     dsu.clear();     dsu_sz.clear();     return mv; } int main() {     vector<int> v = {100, 4, 200, 1, 3, 2};     cout << solve(v);     return 0; }
